# ListRow Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a flexible `ListRow` UI primitive - leading slot, title, subtitle, right text, trailing slot - where the leading slot accepts any `ReactNode` (icon, checkbox, radio).

**Architecture:** Presentational component in `src/components/ui/`, matching the existing primitives (`cn`, named export, concrete Tailwind colors). The one piece of real logic - picking the row height from `size` + whether a subtitle exists - is extracted as a pure helper so it can be smoke-tested in the existing zero-dep harness (which is pure-function only; there is no DOM/React test setup, don't add one).

**Tech Stack:** React (RSC-compatible, no client hooks needed), Tailwind, `lucide-react` for story icons, `@storybook/nextjs`, `tests/smoke/run.ts` harness.

**Design ref:** `docs/plans/2026-06-14-list-row-design.md`. Grouping (`ListRowGroup`) is explicitly **out of scope** - radio groups are a consumer for-loop for now.

```ts
// ponytail: lookup table, not a formula - 4 cases, table is clearest.
export type RowSize = "sm" | "md";

export function rowHeightClass(size: RowSize, hasSubtitle: boolean): string {
  return { md: hasSubtitle ? "h-14" : "h-10", sm: hasSubtitle ? "h-12" : "h-8" }[size];
}
```

### Task 2: ListRow component

**Files:**
- Create: `src/components/ui/list-row.tsx`

No test - presentational, and the harness can't render React. Height logic is already covered by Task 1.

**Step 1: Write the component**

`src/components/ui/list-row.tsx`:
```tsx
import { cn } from "@/lib/utils";
import { rowHeightClass, type RowSize } from "@/lib/list-row";

interface ListRowProps {
  title: string;
  subtitle?: string;
  rightText?: string;
  leading?: React.ReactNode;
  trailing?: React.ReactNode;
  selected?: boolean;
  disabled?: boolean;
  size?: RowSize;
  onClick?: () => void;
  className?: string;
}

export function ListRow({
  title, subtitle, rightText, leading, trailing,
  selected = false, disabled = false, size = "sm", onClick, className,
}: ListRowProps) {
  const interactive = !!onClick;
  return (
    <div
      onClick={disabled ? undefined : onClick}
      className={cn(
        "flex items-center gap-2 pl-1 pr-2 transition-colors",
        rowHeightClass(size, !!subtitle),
        interactive && "group cursor-pointer hover:bg-gray-50",
        selected && "bg-gray-100",
        disabled && "opacity-50 pointer-events-none",
        className,
      )}
    >
      {leading && <div className="flex shrink-0 items-center gap-1">{leading}</div>}

      <div className="min-w-0 flex-1">
        <div className={cn("truncate text-sm text-gray-600", interactive && "group-hover:text-gray-900")}>
          {title}
        </div>
        {subtitle && <div className="truncate text-xs text-gray-400">{subtitle}</div>}
      </div>

      {rightText && <div className="shrink-0 text-xs text-gray-400">{rightText}</div>}

      {trailing && (
        // stop row onClick firing when a trailing action is clicked
        <div className="flex shrink-0 items-center gap-1" onClick={(e) => e.stopPropagation()}>
          {trailing}
        </div>
      )}
    </div>
  );
}
```

Note: the leading slot deliberately does **not** stop propagation - a row click should toggle a checkbox/radio inside it.

### Task 3: Storybook stories

**Files:**
- Create: `src/components/ui/list-row.stories.tsx`

**Step 1: Write the stories**

`leading`/`trailing` are `ReactNode` - not Storybook-controllable. Use string proxy args mapped to nodes in `render`.

```tsx
import type { Meta, StoryObj } from "@storybook/nextjs";
import { File, MoreHorizontal } from "lucide-react";
import { ListRow } from "./list-row";

type Args = React.ComponentProps<typeof ListRow> & {
  leadingKind?: "none" | "icon" | "checkbox" | "radio";
  trailingKind?: "none" | "icon";
};

function leadingFor(k: Args["leadingKind"]) {
  switch (k) {
    case "icon": return <File className="h-4 w-4 text-gray-500" />;
    case "checkbox": return <input type="checkbox" defaultChecked />;
    case "radio": return <input type="radio" name="demo" defaultChecked />;
    default: return null;
  }
}

const meta = {
  title: "UI/ListRow",
  component: ListRow,
  parameters: { layout: "padded" },
  args: {
    title: "Acme Corp", subtitle: "3 decisions", rightText: "2d ago",
    size: "sm", selected: false, disabled: false,
    leadingKind: "icon", trailingKind: "none",
  },
  argTypes: {
    size: { control: "inline-radio", options: ["sm", "md"] },
    leadingKind: { control: "inline-radio", options: ["none", "icon", "checkbox", "radio"] },
    trailingKind: { control: "inline-radio", options: ["none", "icon"] },
    selected: { control: "boolean" },
    disabled: { control: "boolean" },
  },
  render: ({ leadingKind, trailingKind, ...args }: Args) => (
    <div className="w-80 rounded border border-gray-200">
      <ListRow
        {...args}
        leading={leadingFor(leadingKind)}
        trailing={trailingKind === "icon" ? <MoreHorizontal className="h-4 w-4 text-gray-400" /> : null}
      />
    </div>
  ),
} satisfies Meta<Args>;

export default meta;
type Story = StoryObj<typeof meta>;

// Fully configurable from the controls panel.
export const Playground: Story = {};

// Radio group = consumer maps over ListRow with a shared name. No wrapper component (yet).
export const RadioGroup: Story = {
  render: () => {
    const opts = [
      { id: "a", name: "Alice" },
      { id: "b", name: "Bob" },
      { id: "c", name: "Carol" },
    ];
    return (
      <div className="w-80 divide-y divide-gray-100 rounded border border-gray-200">
        {opts.map((o) => (
          <ListRow
            key={o.id}
            title={o.name}
            leading={<input type="radio" name="owner" value={o.id} defaultChecked={o.id === "a"} />}
          />
        ))}
      </div>
    );
  },
};
```
